CREATE PROCEDURE [dbo].[EditarPlan]
	@IdUsuario int,
	@ID_PLAN int,
	@NOMBRE_PLAN varchar(100),
	@DESCRIPCION_PLAN varchar(200),
	@CAPITAL_OBJETIVO_PLAN decimal(10,2),
	@FECHA_INICIO_PLAN datetime,
	@FECHA_FIN_PLAN datetime
AS
BEGIN
	SET NOCOUNT ON;
	UPDATE [dbo].[PLAN]
	SET 
	NOMBRE_PLAN = @NOMBRE_PLAN,
	DESCRIPCION_PLAN = @DESCRIPCION_PLAN,
	CAPITAL_OBJETIVO_PLAN = @CAPITAL_OBJETIVO_PLAN,
	FECHA_INICIO_PLAN = @FECHA_INICIO_PLAN,
	FECHA_FIN_PLAN = @FECHA_FIN_PLAN
	WHERE ID_PLAN = @ID_PLAN
	AND ID_USUARIO = @IdUsuario
END
