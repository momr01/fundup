		AND g.ES_PLAN = 0 )

		AND g.ESTA_ACTIVO_GASTO = 1
		AND g.ES_PLAN = 0
		))
		 ELSE 0

	WHERE i.ID_USUARIO = @idUsuario
	AND i.ESTA_ACTIVO_INGRESO = 1
	AND i.ES_PLAN = 0
	)

(SELECT SUM(g.IMPORTE_GASTO)

 )
