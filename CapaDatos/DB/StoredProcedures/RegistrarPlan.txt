CREATE PROCEDURE [dbo].[RegistrarPlan]
	@ID_USUARIO int,
	@NOMBRE_PLAN varchar(100),
	@DESCRIPCION_PLAN varchar(200),
	@CAPITAL_OBJETIVO_PLAN decimal(10,2),
	@FECHA_INICIO_PLAN datetime,
	@FECHA_FIN_PLAN datetime,
	@FECHA_COMPLETO datetime,
	@ESTA_ACTIVO_PLAN bit = 1
AS
BEGIN
	SET NOCOUNT ON;
	INSERT INTO [dbo].[PLAN]
	([ID_USUARIO],
	[NOMBRE_PLAN],
	[DESCRIPCION_PLAN],
	[CAPITAL_OBJETIVO_PLAN],
	[FECHA_INICIO_PLAN],
	[FECHA_FIN_PLAN],
	[FECHA_COMPLETO],
	[ESTA_ACTIVO_PLAN])
VALUES (
@ID_USUARIO,
@NOMBRE_PLAN,
@DESCRIPCION_PLAN,
@CAPITAL_OBJETIVO_PLAN,
@FECHA_INICIO_PLAN,
@FECHA_FIN_PLAN,
@FECHA_COMPLETO,
@ESTA_ACTIVO_PLAN)
select SCOPE_IDENTITY()
END
