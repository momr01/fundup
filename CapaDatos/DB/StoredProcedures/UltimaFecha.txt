CREATE PROCEDURE [dbo].[UltimaFecha]
	@idUsuario int
AS
BEGIN
	SELECT i.ID_USUARIO, GREATEST(MAX(CAST(i.FECHA_INGRESO AS date)), MAX(CAST(g.FECHA_GASTO AS date))) AS ULTIMA_FECHA
	FROM INGRESO i
	FULL OUTER JOIN GASTO g ON i.ID_USUARIO = g.ID_USUARIO
	WHERE i.ID_USUARIO = @idUsuario
	AND ESTA_ACTIVO_GASTO = 1
	AND ESTA_ACTIVO_INGRESO = 1
	GROUP BY i.ID_USUARIO
END
