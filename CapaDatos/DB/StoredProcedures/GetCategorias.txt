CREATE PROCEDURE [dbo].[GetCategorias]
@idCategoria int = null
AS
BEGIN
SET NOCOUNT ON;
select ID_CATEGORIA AS ID, NOMBRE_CATEGORIA AS NOMBRE, DESCRIPCION_CATEGORIA AS DESCRIPCION, FCREACION_CATEGORIA AS 'FECHA DE CREACIÃ“N', ESTA_ACTIVA_CATEGORIA AS ACTIVA
from CATEGORIA
where ID_CATEGORIA = @idCategoria or @idCategoria is null
AND ESTA_ACTIVA_CATEGORIA = 1
END
